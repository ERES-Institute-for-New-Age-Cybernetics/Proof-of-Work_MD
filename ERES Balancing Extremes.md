`import numpy as np`

*`# ============================================================`*  
*`# 10/10 CONSTRUCT: Comfort = f(Procreation, Profit)`*

`def sigmoid(x):`  
    `"""Bounded output to [0,1]"""`  
    `return 1 / (1 + np.exp(-x))`

`def hill(x, k):`  
    `"""Saturating function: maps effort to [0,1]"""`  
    `return x / (x + k)`

`def comfort(t_r, t_f, params):`  
    `"""`  
    `Full 10/10 comfort function with time constraint.`  
      
    `Parameters:`  
    `t_r, t_f : time allocated to procreation and profit (fractions of 1)`  
    `params   : dictionary of all model parameters`  
      
    `Returns:`  
    `C : comfort score in [0,1]`  
    `"""`  
      
    `# Unpack parameters`  
    `a_r = params['a_r']          # productivity: procreation`  
    `a_f = params['a_f']          # productivity: profit`  
    `k_r = params['k_r']          # half-saturation: procreation`  
    `k_f = params['k_f']          # half-saturation: profit`  
    `alpha = params['alpha']      # max comfort from procreation alone`  
    `beta = params['beta']        # max comfort from profit alone`  
    `gamma = params['gamma']      # max leverage effect`  
    `lam = params['lambda']       # leverage saturation`  
    `delta = params['delta']      # imbalance penalty`  
    `theta_r = params['theta_r']  # subsistence threshold: procreation`  
    `theta_f = params['theta_f']  # subsistence threshold: profit`  
    `eta = params['eta']          # subsistence bonus`  
    `mu = params['mu']            # shift constant`  
      
    `# Convert time to effective levels (with saturation)`  
    `p_r = hill(a_r * t_r, k_r)   # procreation level [0,1]`  
    `p_f = hill(a_f * t_f, k_f)   # profit level [0,1]`  
      
    `# Baseline comfort`  
    `baseline = alpha * p_r + beta * p_f`  
      
    `# Leverage (cross-term with saturation)`  
    `cross = gamma * (p_r * p_f) / (p_r * p_f + lam)`  
      
    `# Imbalance penalty (penalizes extremes)`  
    `penalty = -delta * (p_r - p_f)**2`  
      
    `# Subsistence threshold bonus`  
    `subsistence = eta * (1 if p_r > theta_r and p_f > theta_f else 0)`  
      
    `# Total unbounded score`  
    `raw_score = baseline + cross + penalty + subsistence - mu`  
      
    `# Bound to [0,1] via sigmoid`  
    `C = sigmoid(raw_score)`  
      
    `return C`

`def objective(x, params):`  
    `"""Negative comfort (for minimization)"""`  
    `t_r, t_f = x`

    `return -comfort(t_r, t_f, params)`

`import numpy as np`  
`import matplotlib.pyplot as plt`  
`from scipy.optimize import minimize, Bounds`  
`from mpl_toolkits.mplot3d import Axes3D`

*`# ---- Define model ----`*  
`def sigmoid(x):`  
    `return 1 / (1 + np.exp(-x))`

`def hill(x, k):`  
    `return x / (x + k)`

`def comfort(t_r, t_f, params):`  
    `a_r, a_f = params['a_r'], params['a_f']`  
    `k_r, k_f = params['k_r'], params['k_f']`  
    `alpha, beta = params['alpha'], params['beta']`  
    `gamma, lam = params['gamma'], params['lambda']`  
    `delta = params['delta']`  
    `theta_r, theta_f = params['theta_r'], params['theta_f']`  
    `eta, mu = params['eta'], params['mu']`  
      
    `p_r = hill(a_r * t_r, k_r)`  
    `p_f = hill(a_f * t_f, k_f)`  
      
    `baseline = alpha * p_r + beta * p_f`  
    `cross = gamma * (p_r * p_f) / (p_r * p_f + lam)`  
    `penalty = -delta * (p_r - p_f)**2`  
    `subsistence = eta * (1 if p_r > theta_r and p_f > theta_f else 0)`  
      
    `return sigmoid(baseline + cross + penalty + subsistence - mu)`

`def objective(x, params):`  
    `return -comfort(x[0], x[1], params)`

*`# ---- Parameters ----`*  
`params = {`  
    `'a_r': 0.8, 'a_f': 1.2,`  
    `'k_r': 0.3, 'k_f': 0.4,`  
    `'alpha': 2.0, 'beta': 1.8,`  
    `'gamma': 3.0, 'lambda': 0.2,`  
    `'delta': 2.5,`  
    `'theta_r': 0.4, 'theta_f': 0.4,`  
    `'eta': 1.5,`  
    `'mu': 1.0`  
`}`

*`# ---- Optimize ----`*  
`bounds = Bounds([0, 0], [1, 1])`  
`constraint = {'type': 'ineq', 'fun': lambda x: 1 - x[0] - x[1]}`  
`result = minimize(objective, [0.5, 0.5], args=(params,),`  
                  `method='SLSQP', bounds=bounds, constraints=constraint)`  
`t_r_opt, t_f_opt = result.x`  
`C_opt = -result.fun`

*`# ---- Print results ----`*  
`print(f"Optimal: t_r={t_r_opt:.3f}, t_f={t_f_opt:.3f}, C={C_opt:.4f}")`

## **STAGE 3: Add Leverage (Cross-Term)**

`python`

`"""`  
`STAGE 3: Add Leverage Effect`  
`- Cross-term: γ · H(t_r) · H(t_f)`  
`- Shows synergy between procreation and profit`  
`"""`

`def comfort_with_leverage(t_r, t_f, params):`  
    `"""`  
    `Model with leverage: C = α·H(t_r) + β·H(t_f) + γ·H(t_r)·H(t_f)`  
    `"""`  
    `a_r, a_f = params['a_r'], params['a_f']`  
    `k_r, k_f = params['k_r'], params['k_f']`  
    `alpha, beta = params['alpha'], params['beta']`  
    `gamma = params.get('gamma', 1.0)`  
    `lam = params.get('lambda', 0.5)`  
      
    `p_r = hill(a_r * t_r, k_r)`  
    `p_f = hill(a_f * t_f, k_f)`  
      
    `baseline = alpha * p_r + beta * p_f`  
    `leverage = gamma * (p_r * p_f) / (p_r * p_f + lam)  # Saturated cross-term`  
      
    `raw = baseline + leverage - params.get('mu', 1.0)`  
    `return sigmoid(raw)`

`- Requires minimum level of BOTH for bonus`

`def comfort_with_subsistence(t_r, t_f, params):`  
    `"""`  
    `Full model with subsistence threshold bonus`  
    `"""`  
    `a_r, a_f = params['a_r'], params['a_f']`  
    `k_r, k_f = params['k_r'], params['k_f']`  
    `alpha, beta = params['alpha'], params['beta']`  
    `gamma = params.get('gamma', 1.0)`  
    `lam = params.get('lambda', 0.5)`  
    `delta = params.get('delta', 1.0)`  
    `theta_r = params.get('theta_r', 0.3)`  
    `theta_f = params.get('theta_f', 0.3)`

    `p_r = hill(a_r * t_r, k_r)`

## **STAGE 7: Full 10/10 Construct \- Complete Implementation**

`python`

`"""`

`- Leverage (saturated cross-term)`  
`- Imbalance penalty`  
`- Subsistence threshold`  
`- Asymmetric parameters`  
`- Bounded output (sigmoid)`  
`- Time constraint`  
`"""`

`class ComfortModel10:`  
    `"""Complete 10/10 comfort model"""`  
      
    `def __init__(self, params):`  
        `"""`  
        `Initialize with parameters`  
          
        `Required params:`  
        `a_r, a_f: productivity`  
        `k_r, k_f: half-saturation constants`  
        `alpha, beta: max comfort from each activity`  
        `gamma: max leverage`  
        `lambda: leverage saturation`  
        `delta: imbalance penalty strength`  
        `theta_r, theta_f: subsistence thresholds`  
        `eta: subsistence bonus`  
        `mu: shift constant`  
        `"""`  
        `self.params = params`  
        `self._validate_params()`  
      
    `def _validate_params(self):`  
        `"""Validate all parameters are positive and reasonable"""`  
        `for key, val in self.params.items():`  
            `if key in ['theta_r', 'theta_f']:`  
                `assert 0 < val < 1, f"{key} must be in (0,1)"`  
            `else:`  
                `assert val > 0, f"{key} must be positive"`  
      
    `def p_r(self, t_r):`  
        `"""Effective procreation level"""`  
        `return hill(self.params['a_r'] * t_r, self.params['k_r'])`  
      
    `def p_f(self, t_f):`  
        `"""Effective profit level"""`  
        `return hill(self.params['a_f'] * t_f, self.params['k_f'])`  
      
    `def raw_score(self, t_r, t_f):`  
        `"""Unbounded raw comfort score"""`  
        `p = self.p_r(t_r)`  
        `q = self.p_f(t_f)`  
          
        `# Components`  
        `baseline = self.params['alpha'] * p + self.params['beta'] * q`  
        `leverage = self.params['gamma'] * (p * q) / (p * q + self.params['lambda'])`  
        `penalty = -self.params['delta'] * (p - q)**2`  
        `subsistence = self.params['eta'] * (1 if (p > self.params['theta_r'] and`   
                                                   `q > self.params['theta_f']) else 0)`  
          
        `return baseline + leverage + penalty + subsistence - self.params['mu']`  
      
    `def comfort(self, t_r, t_f):`  
        `"""Bounded comfort score in (0,1)"""`  
        `if t_r < 0 or t_f < 0 or t_r + t_f > 1:`  
            `return np.nan`  
        `return sigmoid(self.raw_score(t_r, t_f))`  
      
    `def optimize(self, method='SLSQP'):`  
        `"""Find optimal time allocation"""`  
        `def objective(x):`  
            `return -self.comfort(x[0], x[1])`  
          
        `bounds = Bounds([0, 0], [1, 1])`  
        `constraint = {'type': 'ineq', 'fun': lambda x: 1 - x[0] - x[1]}`  
          
        `# Try multiple starting points`  
        `best_x = None`  
        `best_f = -np.inf`  
          
        `for start in [(0.2, 0.2), (0.5, 0.5), (0.8, 0.1), (0.1, 0.8)]:`  
            `result = minimize(objective, start, method=method,`  
                            `bounds=bounds, constraints=constraint)`  
            `if result.success and -result.fun > best_f:`  
                `best_f = -result.fun`  
                `best_x = result.x`  
          
        `return best_x[0], best_x[1], best_f`  
      
    `def evaluate_extremes(self):`  
        `"""Test pure extreme scenarios"""`  
        `return {`  
            `'all_procreation': self.comfort(1.0, 0.0),`  
            `'all_profit': self.comfort(0.0, 1.0),`  
            `'balanced_50_50': self.comfort(0.5, 0.5)`  
        `}`
